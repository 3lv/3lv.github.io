#!/bin/bash # curl https://3lv.github.io | sudo sh github="https://github.com/3lv" dotfiles="${github}/dot" nvim="${github}/nvim" cyan="\033[36m" green="\033[32m" reset="\033[0m" # Installing dotfiles echo -e "${cyan}Installing dotfiles ${green}[1/3]${reset}" cd ~ rm -rf .git git init >/dev/null 2>&1 git remote add origin $dotfiles git fetch origin master git reset --hard origin/master # Install system applications (from dotfiles/.install.sh) echo -e "${cyan}Installing system applications ${green}[2/3]${reset}" chmod u+x ~/.install.sh sudo ~/.install.sh # Install nvim config + plugins echo -e "${cyan}Installing nvim config + plugins ${green}[3/3]${reset}" rm -rf ~/.config/nvim mkdir -p ~/.config git clone $nvim ~/.config/nvim nvim --headless +Setup >/dev/null echo -e "${cyan}DONE!${reset}"
